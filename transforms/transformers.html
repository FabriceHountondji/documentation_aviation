 13def count_icao24_in_dirs(directories, output_file):
 14    """Counts occurrences of ICAO24 addresses in CSV files within specified directories.
 15
 16    Args:
 17        directories (list[str]): List of directory patterns or paths to search for CSV files.
 18        output_file (str): Path to the output CSV file where results will be saved.
 19    """
 20    icao_counter = Counter()
 21
 22    # Parcourt chaque motif ou répertoire de fichiers
 23    for pattern in directories:
 24        csv_files = glob.glob(pattern, recursive=True)
 25        print(f"{len(csv_files)} fichiers trouvés avec le motif {pattern}")
 26
 27        for file_path in csv_files:
 28            if not file_path.lower().endswith(".csv"):
 29                continue
 30
 31            print(f"Lecture de {file_path}...")
 32
 33            try:
 34                with open(file_path, mode="r", encoding="utf-8-sig") as f:
 35                    reader = csv.DictReader(f)
 36
 37                    # Vérifie la présence de la colonne 'icao24'
 38                    if not reader.fieldnames or 'icao24' not in reader.fieldnames:
 39                        print(f"Le fichier {file_path} ne contient pas la colonne 'icao24'.")
 40                        continue
 41
 42                    # Compte les occurrences d'icao24
 43                    for row in reader:
 44                        icao = row.get('icao24')
 45                        if icao and icao.strip():
 46                            icao_counter[icao.strip()] += 1
 47
 48            except Exception as e:
 49                print(f"Erreur lecture {file_path}: {e}")
 50
 51    # Trie les résultats par nombre d'occurrences décroissant
 52    sorted_icao = sorted(icao_counter.items(), key=lambda x: x[1], reverse=True)
 53
 54    # Écrit le fichier résultat
 55    os.makedirs(os.path.dirname(output_file), exist_ok=True)
 56    with open(output_file, mode="w", newline="", encoding="utf-8") as out:
 57        writer = csv.writer(out)
 58        writer.writerow(["icao24", "compte"])
 59        for icao, count in sorted_icao:
 60            writer.writerow([icao, count])
 61
 62    print(f"\nFichier résultat généré : {output_file}")
 63    print(f"Total unique ICAO24 trouvés : {len(icao_counter)}")

 66def convert_dates_in_dirs(directories):
 67    """Converts date columns in CSV files to ISO-8601 formatted strings with Montreal timezone.
 68
 69    Args:
 70        directories (list[str]): List of directory paths containing CSV files to process.
 71    """
 72    # --- Fuseau horaire Montréal ---
 73    montreal_tz = pytz.timezone("America/Toronto")
 74
 75    # --- Parcours des fichiers CSV ---
 76    for dir_path in directories:
 77        for file_name in os.listdir(dir_path):
 78            if file_name.endswith(".csv"):
 79                file_path = os.path.join(dir_path, file_name)
 80                print(f"Traitement du fichier : {file_path}")
 81                
 82                # Lire le CSV
 83                df = pd.read_csv(file_path)
 84                
 85                # Vérifier si la colonne 'date' existe
 86                if 'date' in df.columns:
 87                    # Convertir la colonne 'date' en datetime et appliquer le fuseau horaire
 88                    df['date'] = pd.to_datetime(df['date'], errors='coerce')
 89                    
 90                    # Si date est juste YYYY-MM-DD, ajouter minuit comme heure
 91                    df['date'] = df['date'].apply(lambda x: datetime.combine(x, datetime.min.time()) if pd.notnull(x) else x)
 92                    
 93                    # Localiser au fuseau de Montréal et formatter en ISO-8601 avec offset -0400
 94                    df['date'] = df['date'].apply(lambda x: montreal_tz.localize(x).strftime('%Y-%m-%dT%H:%M:%S%z') if pd.notnull(x) else x)
 95                    
 96                    # Sauvegarder le CSV mis à jour (écrase l'original ou crée un nouveau fichier)
 97                    df.to_csv(file_path, index=False)
 98                    print(f"Fichier mis à jour : {file_path}")
 99                else:
100                    print(f"Colonne 'date' absente dans {file_path}")

103def enrich_flights_with_icao(registration_dirs, flight_dirs_or_files):
104    """Enriches flight CSV files with ICAO24 addresses based on aircraft registrations.
105
106    Args:
107        registration_dirs (list[str]): Directories containing mapping files between registrations and ICAO24.
108        flight_dirs_or_files (str | Path | list): Paths to flight CSV files or directories containing them.
109    """
110
111    # Charger les correspondances registration → icao24
112    registration_to_icao = {}
113
114    for reg_dir in registration_dirs:
115        if not os.path.exists(reg_dir):
116            print(f"Répertoire introuvable : {reg_dir}")
117            continue
118
119        for file in os.listdir(reg_dir):
120            if not file.endswith(".csv"):
121                continue
122
123            path = os.path.join(reg_dir, file)
124            with open(path, newline='', encoding="utf-8") as f:
125                reader = csv.DictReader(f)
126                for row in reader:
127                    reg = (row.get("registration") or "").strip().upper()
128                    icao = (row.get("icao24") or "").strip()
129                    if reg and icao:
130                        registration_to_icao[reg] = icao
131
132    print(f"{len(registration_to_icao)} correspondances registration → icao24 chargées depuis les répertoires.")
133
134    # Normaliser l'entrée : liste de chemins
135    if isinstance(flight_dirs_or_files, (str, Path)):
136        flight_dirs_or_files = [flight_dirs_or_files]
137
138    for item in flight_dirs_or_files:
139        path = Path(item)
140
141        # ------------------------
142        #  CAS A : fichier CSV
143        # ------------------------
144        if path.is_file() and path.suffix == ".csv":
145            flight_files = [str(path)]
146            print(f"\nMise à jour du fichier : {path}")
147
148        # ------------------------
149        #  CAS B : répertoire
150        # ------------------------
151        elif path.is_dir():
152            print(f"\nMise à jour des fichiers dans : {path}")
153            flight_files = [
154                str(p) for p in path.glob("*.csv")
155            ]
156        else:
157            print(f"Chemin ignoré (ni fichier CSV ni dossier valide) : {path}")
158            continue
159
160        if not flight_files:
161            print("Aucun fichier CSV trouvé.")
162            continue
163
164        # ------------------------
165        #  Traitement des fichiers
166        # ------------------------
167        for flight_file in flight_files:
168            enriched_rows = []
169            enriched_count = 0
170
171            with open(flight_file, newline='', encoding="utf-8") as f:
172                reader = csv.DictReader(f)
173                fieldnames = reader.fieldnames or []
174
175                # Ajouter la colonne ICAO24 si absente
176                if "icao24" not in fieldnames:
177                    fieldnames.append("icao24")
178
179                for row in reader:
180                    if not isinstance(row, dict):
181                        continue
182
183                    reg = (row.get("registration") or "").strip().upper()
184                    if reg in registration_to_icao:
185                        row["icao24"] = registration_to_icao[reg]
186                        enriched_count += 1
187                    else:
188                        val = row.get("icao24")
189                        row["icao24"] = val.strip() if isinstance(val, str) else ""
190
191                    enriched_rows.append(row)
192
193            # réécriture du fichier
194            with open(flight_file, "w", newline='', encoding="utf-8") as f:
195                writer = csv.DictWriter(f, fieldnames=fieldnames, quoting=csv.QUOTE_MINIMAL)
196                writer.writeheader()
197                writer.writerows(enriched_rows)
198
199            print(f"{enriched_count} ICAO24 ajoutés → {os.path.basename(flight_file)}")

269def update_flights_with_companies(compagnies_csv, flights_dir):
270    """Updates flight CSV files with airline names based on callsign prefixes.
271
272    Args:
273        compagnies_csv (str | Path): Path to the companies CSV file.
274        flights_dir (str | Path): Directory containing flight CSV files to update.
275
276    Raises:
277        FileNotFoundError: If the companies CSV file does not exist.
278        NotADirectoryError: If the flights directory does not exist.
279    """
280    compagnies_csv = Path(compagnies_csv)
281    flights_dir = Path(flights_dir)
282
283    # Vérifications
284    if not compagnies_csv.is_file():
285        raise FileNotFoundError(f"Le fichier compagnies n'existe pas : {compagnies_csv}")
286
287    if not flights_dir.is_dir():
288        raise NotADirectoryError(f"Le dossier flights_meanTime n'existe pas : {flights_dir}")
289
290    # Charger compagnies
291    df_compagnies = pd.read_csv(compagnies_csv)
292    df_compagnies = df_compagnies.rename(columns={"company": "company_name"})
293
294    for file in flights_dir.glob("*.csv"):
295        print(f"Mise à jour de : {file}")
296
297        df_flights = pd.read_csv(file)
298
299        df_flights["callsign_prefix"] = df_flights["callsign"].astype(str).str[:3]
300
301        df_merged = df_flights.merge(
302            df_compagnies[["letter_id", "company_name"]],
303            left_on="callsign_prefix",
304            right_on="letter_id",
305            how="left"
306        )
307
308        # FIX IMPORTANT : mise à jour airline uniquement si vide ou NaN
309        df_merged["airline"] = df_merged["airline"].mask(
310            (df_merged["airline"].isna()) |
311            (df_merged["airline"].astype(str).str.strip() == ""),
312            df_merged["company_name"]
313        )
314
315        # Nettoyage
316        df_merged = df_merged.drop(columns=["callsign_prefix", "letter_id", "company_name"])
317
318        # Sauvegarde dans le fichier d'origine
319        df_merged.to_csv(file, index=False)
320
321    print("Mise à jour terminée.")

324def calculate_flight_time(directories):
325    """Calculates flight duration in minutes for CSV files in specified directories.
326
327    Args:
328        directories (list[str]): List of directory paths containing flight CSV files.
329    """
330    for directory in directories:
331        for filename in os.listdir(directory):
332            if filename.endswith(".csv"):
333                filepath = os.path.join(directory, filename)
334                print(f"Traitement : {filepath}")
335
336                # Lecture du CSV
337                df = pd.read_csv(filepath)
338
339                # Vérification des colonnes (sécurité)
340                if "firstSeen" not in df.columns or "lastSeen" not in df.columns:
341                    print(f"Colonnes manquantes, fichier ignoré : {filename}")
342                    continue
343
344                # Conversion des colonnes en datetime
345                df['firstSeen'] = pd.to_datetime(df['firstSeen'],errors='coerce', utc=True)
346                df['lastSeen'] = pd.to_datetime(df['lastSeen'],errors='coerce', utc=True)
347
348                # Ignorer les lignes où firstSeen ou lastSeen est manquant
349                df = df[df["firstSeen"].notna() & df["lastSeen"].notna()]
350
351                # Calcul du temps de vol en minutes
352                df['temps_vol'] = (df['lastSeen'] - df['firstSeen']).dt.total_seconds() / 60
353                df['temps_vol'] = df['temps_vol'].round(2)  # Arrondi optionnel
354
355                # Réécriture dans le même fichier
356                df.to_csv(filepath, index=False)
357
358                print(f"Mise à jour effectuée : {filepath}")

361def compute_mean_flight_times(directories, output_prefix="result_mean_times"):
362    """Computes mean flight times per (callsign, departure, arrival) across multiple directories.
363
364    Args:
365        directories (list[str]): List of directories containing merged CSV files.
366        output_prefix (str): Prefix for the generated output filename. Defaults to "result_mean_times".
367    """
368
369    # ================================
370    # Préparation nom fichier sortie
371    # ================================
372    montreal_tz = pytz.timezone("America/Toronto")
373    timestamp = datetime.now(montreal_tz).strftime("%Y%m%d_%H%M%S")
374
375    output_dir = "datasets/flights_meanTime"
376    os.makedirs(output_dir, exist_ok=True)
377
378    output_file = os.path.join(
379        output_dir,
380        f"{output_prefix}_{timestamp}.csv"
381    )
382
383    # ================================
384    # Récupération de tous les CSV
385    # ================================
386    all_files = []
387
388    for directory in directories:
389        if not os.path.isdir(directory):
390            print(f"Répertoire introuvable : {directory}")
391            continue
392
393        for file in os.listdir(directory):
394            if file.endswith(".csv"):
395                print(f"Fichier détecté : {os.path.join(directory, file)}")
396                all_files.append(os.path.join(directory, file))
397
398    if not all_files:
399        print("Aucun fichier CSV trouvé.")
400        return
401
402    print(f"{len(all_files)} fichiers détectés.")
403    
404
405    # ================================
406    # Lecture & concaténation
407    # ================================
408    dataframes = []
409
410    for file in all_files:
411        try:
412            df = pd.read_csv(file)
413            dataframes.append(df)
414        except Exception as e:
415            print(f"Erreur lecture {file}: {e}")
416
417    print(f"{len(dataframes)} fichiers chargés avec succès.")
418    if not dataframes:
419        print("Impossible de charger les fichiers.")
420        return
421
422    all_data = pd.concat(dataframes, ignore_index=True)
423
424    print(f"Total lignes combinées : {len(all_data)}")
425
426    # ================================
427    # Normalisation colonnes
428    # ================================
429    required_cols = [
430        "callsign_opensky",
431        "departure_airport",
432        "arrival_airport",
433        "temps_vol",
434        "airline",
435    ]
436
437    for col in required_cols:
438        if col not in all_data.columns:
439            raise ValueError(f"Colonne manquante dans les données : {col}")
440
441    # Renommer pour correspondre au fichier résultat attendu
442    all_data = all_data.rename(columns={
443        "callsign_opensky": "callsign",
444        "departure_airport": "iata_departure",
445        "arrival_airport": "iata_arrival",
446    })
447
448    # ================================
449    # Nettoyage
450    # ================================
451    all_data = all_data.dropna(subset=[
452        "callsign",
453        "iata_departure",
454        "iata_arrival",
455        "temps_vol"
456    ])
457
458    print(f"Lignes après nettoyage : {len(all_data)}")
459
460    # S'assurer que temps_vol est numérique
461    all_data["temps_vol"] = pd.to_numeric(all_data["temps_vol"], errors="coerce")
462    all_data = all_data.dropna(subset=["temps_vol"])
463
464    print(f"Lignes après validation temps_vol : {len(all_data)}")
465    # ================================
466    # Agrégation UNIQUE par vol
467    # ================================
468    result = (
469        all_data
470        .groupby(
471            ["callsign", "iata_departure", "iata_arrival", "airline"],
472            as_index=False
473        )["temps_vol"]
474        .mean()
475        .rename(columns={"temps_vol": "temps_vol"})
476    )
477
478    print(f"Vols uniques calculés : {len(result)}")
479    # ================================
480    # Sauvegarde
481    # ================================
482    result.to_csv(output_file, index=False, encoding="utf-8")
483
484    print(f"Fichier créé : {output_file}")
485    print(f"{len(result)} vols uniques calculés.")

531def load_flights_from_directories(directories, processed_files, source="fr24"):
532    """Loads flight data from CSV files in multiple directories, skipping already processed ones.
533
534    Args:
535        directories (list[str]): List of directory paths to search.
536        processed_files (set): Set of filenames that should be skipped.
537        source (str): Source identifier (e.g., "fr24", "opensky"). Defaults to "fr24".
538
539    Returns:
540        tuple[list[dict], list[str]]: (List of flight rows as dicts, List of newly used filenames).
541    """
542    flights = []
543    new_files_used = []  # Collecter les nouveaux fichiers traités pour mise à jour
544
545    for directory in directories:
546        if not os.path.exists(directory):
547            print(f"Répertoire introuvable : {directory}")
548            continue
549
550        for file in os.listdir(directory):
551            if not file.endswith(".csv"):
552                continue
553
554            if file in processed_files:
555                print(f"Fichier déjà traité, ignoré : {file}")
556                continue
557
558            filepath = os.path.join(directory, file)
559            with open(filepath, newline='', encoding='utf-8') as f:
560                reader = csv.DictReader(f)
561                count_before = len(flights)
562
563                for row in reader:
564                    row["source_file"] = file
565                    flights.append(row)
566
567                if len(flights) > count_before:
568                    new_files_used.append(file)
569
570        print(f"{len(flights)} vols cumulés après lecture de {directory} ({source})")
571
572    return flights, new_files_used

574def match_flights_arrivals(fr_flights, opensky_flights, output_base, new_files_used, processed_files, processed_files_txt):
575    """Matches FR24 arrivals with OpenSky historical flights based on ICAO24 and time proximity.
576
577    Args:
578        fr_flights (list[dict]): List of flight records from FlightRadar24.
579        opensky_flights (list[dict]): List of flight records from OpenSky.
580        output_base (str): Base filename for the matched results CSV.
581        new_files_used (list[str]): List of newly processed filenames to track.
582        processed_files (set): Set of already processed filenames.
583        processed_files_txt (str): Path to the tracking text file.
584    """
585    if not fr_flights or not opensky_flights:
586        print("Aucun vol à matcher.")
587        return
588
589    matched_rows = []
590    delta_max = timedelta(minutes=10)
591
592    valid_arrival_airports = {"YUL", "YYZ", "YVR", "CDG", "ORY", "LYS", "LAX", "ORD", "JFK", "MIA", "FCO", "MXP", "CTA","MAD", "BCN", "PMI", "FRA", "MUC", "HAM","ALG", "CMN", "TUN","HND", "PEK"}
593
594
595    for fr in fr_flights:
596        fr_icao24 = fr.get("icao24", "").strip().lower()
597        fr_arr_airport = fr.get("arrival_airport_code", "").strip().upper()
598        fr_expected = parse_datetime(fr.get("expected_time", ""))
599
600        if not fr_icao24 or not fr_expected:
601            continue
602
603        for osky in opensky_flights:
604            osky_icao24 = osky.get("icao24", "").strip().lower()
605            osky_arr_iata = osky.get("iata_arrival", "").strip().upper()
606            osky_last_seen = parse_datetime(osky.get("lastSeen", ""))
607
608            if not osky_icao24 or not osky_last_seen:
609                continue
610
611            if osky_icao24 != fr_icao24:
612                continue
613
614            if osky_arr_iata not in valid_arrival_airports:
615                continue
616
617            if fr_arr_airport and osky_arr_iata != fr_arr_airport:
618                continue
619
620            if abs(osky_last_seen - fr_expected) <= delta_max:
621                matched_rows.append({

541    Returns:

554                continue

556            if file in processed_files:

558                continue

560            filepath = os.path.join(directory, file)
561            with open(filepath, newline='', encoding='utf-8') as f:
562                reader = csv.DictReader(f)
563                count_before = len(flights)
564
565                for row in reader:
566                    row["source_file"] = file
567                    flights.append(row)
568
569                if len(flights) > count_before:
570                    new_files_used.append(file)
571
572        print(f"{len(flights)} vols cumulés après lecture de {directory} ({source})")
573
574    return flights, new_files_used
