 45    """Converts a Unix timestamp to a formatted string in Montreal time zone.

387def ensure_datetime_series(df, col):
388    """Ensures a DataFrame column is a naive datetime series.
389
390    Args:
391        df (pd.DataFrame): The input DataFrame.
392        col (str): The column name to convert.
393
394    Returns:
395        pd.Series: The converted datetime series without timezone information.
396    """
397    s = df[col]
398
399    # 1) S'assurer que c'est du datetime (timezone ou non)
400    if not pd.api.types.is_datetime64_any_dtype(s):
401        s = pd.to_datetime(s, errors="coerce")
402
403    # 2) Supprimer la timezone si présente
404    if getattr(s.dt, "tz", None) is not None:
405        s = s.dt.tz_convert(None)
406
407    return s

410def compute_delays(df):
411    """Computes delay metrics for flight instances.
412
413    Calculates:
414      - scheduled, expected_time (normalized)
415      - delay_min (actual - scheduled in minutes)
416      - delayed_15 (boolean if delay > 15 minutes)
417
418    Args:
419        df (pd.DataFrame): DataFrame containing scheduled and expected/seen times.
420
421    Returns:
422        pd.DataFrame: DataFrame with added delay columns.
423    """
424    df = df.copy()
425
426    df["scheduled"] = pd.to_datetime(df["scheduled"], errors="coerce")
427    df["expected_time"] = pd.to_datetime(df["expected_time"], errors="coerce")
428
429
430    df = df[df["scheduled"].notna() & df["expected_time"].notna()]
431
432
433    df["scheduled"] = ensure_datetime_series(df, "scheduled")
434    # prioriser expected_time comme 'actual' (ou lastSeen pour arrival)
435    actual = None
436    if "expected_time" in df.columns:
437        actual = ensure_datetime_series(df, "expected_time")
438    if actual is None and "lastSeen" in df.columns:
439        actual = ensure_datetime_series(df, "lastSeen")
440    if actual is None and "firstSeen" in df.columns:
441        actual = ensure_datetime_series(df, "firstSeen")
442
443    df["expected_time"] = actual
444    df["delay_min"] = (df["expected_time"] - df["scheduled"]).dt.total_seconds() / 60.0
445    df["delay_min"] = df["delay_min"].replace([np.inf, -np.inf], np.nan)
446    df["delayed_15"] = df["delay_min"] > 15
447    return df

449def operational_performance_analysis(df_instances, df_flights):
450    """Analyzes operational performance by comparing actual travel times to means.
451
452    Args:
453        df_instances (pd.DataFrame): Flight instances data.
454        df_flights (pd.DataFrame): Reference flight data with mean_time.
455
456    Returns:
457        tuple[pd.DataFrame, pd.DataFrame]: (route_performance_agg, detailed_df)
458    """
459    df = df_instances.copy()
460
461    # st.write("Colonnes df_instances :", df.columns.tolist())
462    # st.write("Taille initiale df_instances :", len(df))
463
464    # ----------------------------------------------------
465    # Calcul du travel_time AVANT filtrage
466    # ----------------------------------------------------
467    if df["travel_time"].isna().all():
468        df["firstSeen"] = pd.to_datetime(df["firstSeen"], errors="coerce", unit="s")
469        df["lastSeen"] = pd.to_datetime(df["lastSeen"], errors="coerce", unit="s")
470
471        df["travel_time"] = (
472            df["lastSeen"] - df["firstSeen"]
473        ).dt.total_seconds() / 60
474
475    # st.write("travel_time non null :", df["travel_time"].notna().sum())
476
477    # ----------------------------------------------------
478    # Nettoyage routes
479    # ----------------------------------------------------
480    df["departure_iata"] = df["departure_iata"].str.strip().str.upper()
481    df["arrival_iata"] = df["arrival_iata"].str.strip().str.upper()
482
483    # ----------------------------------------------------
484    # Drop NA APRÈS calcul
485    # ----------------------------------------------------
486    # df = df.dropna(subset=["travel_time", "departure_iata", "arrival_iata"])
487    # st.write("Après dropna :", len(df))
488
489    # ----------------------------------------------------
490    # Table de référence
491    # ----------------------------------------------------
492    ref = df_flights[[
493        "departure_iata", "arrival_iata", "mean_time"
494    ]].dropna()
495
496    ref["departure_iata"] = ref["departure_iata"].str.strip().str.upper()
497    ref["arrival_iata"] = ref["arrival_iata"].str.strip().str.upper()
498
499    # st.write("Taille df_flights ref :", len(ref))
500
501    # ----------------------------------------------------
502    # Merge
503    # ----------------------------------------------------
504    df = df.merge(
505        ref,
506        on=["departure_iata", "arrival_iata"],
507        how="left"
508    )
509
510    # st.write("Après merge :", len(df))
511    # st.write("mean_time non null :", df["mean_time"].notna().sum())
512
513    # ----------------------------------------------------
514    # Performance
515    # ----------------------------------------------------
516    df["delta_vs_mean"] = df["travel_time"] - df["mean_time"]
517
518    route_perf = (
519        df.groupby(["departure_iata", "arrival_iata"])
520        .agg(
521            n_flights=("travel_time", "count"),
522            mean_travel_time=("travel_time", "mean"),
523            std_travel_time=("travel_time", "std"),
524            mean_reference=("mean_time", "mean"),
525            mean_delta=("delta_vs_mean", "mean"),
526            std_delta=("delta_vs_mean", "std")
527        )
528        .reset_index()
529        .sort_values("std_travel_time", ascending=False)
530    )
531
532    return route_perf, df

537def seasonal_analysis(df_instances):
538    """Performs seasonal analysis (monthly, daily, route-based) on flight instances.
539
540    Args:
541        df_instances (pd.DataFrame): Flight instances data.
542
543    Returns:
544        tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame | None]: (monthly_agg, weekday_agg, route_pivot)
545    """
546    df = df_instances.copy()
547
548    # 1. Conversion robuste en datetime avec gestion des fuseaux horaires (UTC)
549    # On applique utc=True pour harmoniser les différents offsets (+00:00, -05:00)
550    if "expected_time" in df.columns:
551        df["expected_time"] = pd.to_datetime(df["expected_time"], errors="coerce", utc=True)
552    else:
553        raise KeyError("La colonne 'expected_time' est manquante dans df")
554
555    # 2. Gestion du fallback sur 'scheduled' (également converti en UTC)
556    if "scheduled" in df.columns:
557        scheduled_dt = pd.to_datetime(df["scheduled"], errors="coerce", utc=True)
558        df["expected_time"] = df["expected_time"].fillna(scheduled_dt)
559
560    # 3. Nettoyage des lignes sans aucune date valide
561    df = df.dropna(subset=["expected_time"])
562
563    # 4. Extraction des composants temporels
564    df["month"] = df["expected_time"].dt.month
565    df["dayofweek"] = df["expected_time"].dt.day_name()
566
567    # 5. Agrégation mensuelle (on force l'apparition de tous les mois 1 à 12)
568    monthly_counts = df.groupby("month").size()
569    all_months = pd.Index(range(1, 13), name="month")
570    monthly = monthly_counts.reindex(all_months, fill_value=0).rename("count").reset_index()
571
572    # 6. Agrégation par jour de la semaine
573    weekday = df.groupby("dayofweek").size().reindex([
574        'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday', 'Sunday'
575    ], fill_value=0).rename("count").reset_index()
576
577    # 7. Heatmap route x month
578    if "departure_iata" in df.columns and "arrival_iata" in df.columns:
579        # On s'assure d'utiliser l'ID pour le comptage
580        count_col = "id" if "id" in df.columns else df.columns[0]
581        df["route"] = df["departure_iata"].astype(str) + "-" + df["arrival_iata"].astype(str)
582        pivot = df.pivot_table(index="route", columns="month", values=count_col, aggfunc="count", fill_value=0)
583    else:
584        pivot = None
585
586    return monthly, weekday, pivot

677def haversine_distance(lat1, lon1, lat2, lon2):
678    """Calculates the Haversine distance between two points on Earth.
679
680    Args:
681        lat1 (float): Latitude of point 1.
682        lon1 (float): Longitude of point 1.
683        lat2 (float): Latitude of point 2.
684        lon2 (float): Longitude of point 2.
685
686    Returns:
687        float: Distance in kilometers.
688    """
689    # Rayon de la Terre en km
690    R = 6371.0
691    
692    phi1, phi2 = np.radians(lat1), np.radians(lat2)
693    dphi = np.radians(lat2 - lat1)
694    dlambda = np.radians(lon2 - lon1)
695    
696    a = np.sin(dphi / 2.0)**2 + \
697        np.cos(phi1) * np.cos(phi2) * np.sin(dlambda / 2.0)**2
698    
699    return 2 * R * np.arctan2(np.sqrt(a), np.sqrt(1 - a))

679    """Calculates the Haversine distance between two points on Earth.
680
681    Args:
682        lat1 (float): Latitude of point 1.
683        lon1 (float): Longitude of point 1.
684        lat2 (float): Latitude of point 2.
685        lon2 (float): Longitude of point 2.
686
687    Returns:
688        float: Distance in kilometers.
689    """
690    # Rayon de la Terre en km
691    R = 6371.0
692

694    dphi = np.radians(lat2 - lat1)
695    dlambda = np.radians(lon2 - lon1)
696

698        np.cos(phi1) * np.cos(phi2) * np.sin(dlambda / 2.0)**2
699    
700    return 2 * R * np.arctan2(np.sqrt(a), np.sqrt(1 - a))
